19  /**
20   * This enumeration provides constants for all dialects supported by Hibernate.
21   *
22   * The dialects supported by Hibernate can be found in the 
23   * <a href="http://docs.jboss.org/hibernate/orm/4.3/manual/en-US/html_single/#configuration-optional-dialects">
24   * Hibernate documentation</a>.
25   * Also this enumeration provides the convenient method
26   * {@link #getDialectClass()} for getting the class name of the Hibernate
27   * dialect.
28   *
29   * @author <a href="mailto:jens.pelzetter@googlemail.com">Jens Pelzetter</a>
30   */

79      /**
80       * Property for holding the name of the Hibernate dialect class.
81       */
82      private final String dialectClass;
83  
84      /**
85       * Private constructor, used to create the Enum instances for each dialect.
86       *
87       * @param dialectClass The dialect class for the specific dialect.
88       */
89      private Dialect(final String dialectClass) {
90          this.dialectClass = dialectClass;
91      }
92  
93      /**
94       * Getter for the dialect class.
95       *
96       * @return The name of the dialect class, for example
97       *         {@code org.hibernate.dialect.PostgreSQL9Dialect} for
98       *         {@link #POSTGRESQL9}.
99       */
100     public String getDialectClass() {
101         return dialectClass;
102     }
